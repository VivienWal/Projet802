stream {

    log_format tcp '$remote_addr [$time_local] '
                '$protocol $status $bytes_sent '
                '$upstream_addr $upstream_bytes_received '

	access_log /var/log/nginx/relay_access.log tcp;

    }

    }

        proxy_pass node3-key;
